Multiplication, division, and comparison operators

// Write a function that takes two numbers, say a and b, as arguments.
// If a is smaller than b, divide a by b
// Otherwise, multiply both numbers
// Return the resulting value

Example

    myFunction(10, 100)
    myFunction(90, 45)
    myFunction(8, 20)
    myFunction(2, 0.5)

Solution

    0.1
    4050
    0.4
    1

Resolution

function myFunction1(a, b) {
return a < b ? a / b : a * b
}

Clear up the chaos behind these strings

// Can you figure out how to clear up the chaos?
// Write a function that joins these strings together such that they form the words 'Javascript', 'Countryside', and 'Downtown'!
// You might want to apply basic JS string methods such as replace(), split(), slice() etc

Example

    myFunction('java', 'tpi%rcs')
    myFunction('c%ountry', 'edis')
    myFunction('down', 'nw%ot')

Solution

    'Javascript'
    'Countryside'
    'Downtown'

Resolution

function myFunction10(uno, dos) {
uno = uno.replaceAll('%','');
dos = dos.replaceAll('%','');
let newCharUno = '';
let newCharDos = '';
let inicio = dos.length - 1;

for (let i = 0; i < uno.length ; i++ ) {
if (i==0) {
newCharUno = newCharUno + uno.charAt(i).toUpperCase();
} else {
newCharUno = newCharUno + uno.charAt(i);}
}

for (let j = inicio; j > -1 ; j-- ) {
newCharDos = newCharDos + dos.charAt(j);
}
return newCharUno + newCharDos;
}

function myFunction10(a, b) {
const first = a.split('').map(part => part.replace('%', '')).join('');
const second = b.split('').reverse().map(part => part.replace('%', '')).join('');
return first.charAt(0).toUpperCase() + first.slice(1) + second
}

Return the next higher prime number

// This challenge is a little bit more complex
// Write a function that takes a number as argument
// If the number is prime, return the number
// If not, return the next higher prime number

Example

    myFunction(38)
    myFunction(7)
    myFunction(115)
    myFunction(2000)

Solution

    41
    7
    127
    2003

Resolution

function myFunction(num) {
return esPrimo(num)? num : proximoPrimo(num);
function esPrimo(num) {
let cantDivisores = 0;
for (let i = 0; i<=num ; i++) {
(num%i===0)? ++cantDivisores: 0;
}
return (cantDivisores==2||num==1);
}
function proximoPrimo(num) {
while (!esPrimo(num)) {
++num;
}
return num;
}
}

Resolution II

function myFunction( input ) {
function isPrime(num) {
for (let i = 2; i <= Math.sqrt(num); i++) {
if (num % i === 0) return false;
}
return num > 1;
}
let n = input;
while (!isPrime(n)) n++;
return n
}
